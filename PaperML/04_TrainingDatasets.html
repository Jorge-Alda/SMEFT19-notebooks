Random points¶

from tqdm.notebook import tqdm
from parscanning import RandomScan

RS = RandomScan(SMEFT19.likelihood_global, [-0.22, -0.15, -0.03, -0.12, 0],
                [-0.04, 0.15, 0.03, 0.075, 3.5], 10)

for i in tqdm(range(111)):
    RS.run_mp(4, SMEFT19.scenarios.rotBII)
    RS.write('../data/samples/randompoints.dat', 'at')
    RS.clear()
